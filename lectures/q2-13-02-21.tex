\section{Time-independent perturbation theory: degenerate case}

Note: if $\exists$ operator $A$ with 
\begin{dgroup}[]
	\begin{dmath}[]
		\comm{A}{V}=0
	\end{dmath}
	\begin{dsuspend}
		and
	\end{dsuspend}
	\begin{dmath}[]
		A\ket{\psi_{n_i}^{(0)}}=a_{n_i}\ket{\psi_{n_i}^{(0)}}
	\end{dmath},
	\begin{dsuspend}
		with
	\end{dsuspend}
	\begin{dmath}[]
		a_{n_i}\neq a_{n_k}\condition{for $k\neq i$}
	\end{dmath}
	\begin{dsuspend}
		then these $\ket{\psi_{n_i}^{(0)}}$ are ``good'' eigenstates
	\end{dsuspend}
\end{dgroup}
\begin{proof}
	\begin{dgroup}[]
	\begin{dmath}[]
		\sand{\psi_{n_i}^{(0)}}{\comm{A}{V}}{\psi_{n_k}^{(0)}}=0=\underbrace{\left( a_{n_i}-a_{n_k} \right)}_{\neq 0}\underbrace{\sand{\psi_{n_i}^{(0)}}{V}{\psi_{n_k}^{(0)}}}_{V_{ik}}
		\end{dmath}
		\begin{dmath}[]
		\Rightarrow V_{ik}=0
	\end{dmath}
	\end{dgroup}
	\end{proof}
